26double get_density(double xi, double eta, int r)
27{
28 double rt;
29 double xt, yt, jt;
30 int i, j;
31
32 i = r / m;
33 j = r % m;
34 //*********下面获得节点的物理坐标*********//
35 xt = o[i][j].phi_x(xi,eta);
36 yt = o[i][j].phi_y(xi,eta);
37
38 //*********下面用强质量守恒计算密度*******//
39 jt = o[i][j].Jacobi_0(xi,eta);
40 rt = ini_rho(xt,yt) * jt;
41 jt = o[i][j].Jacobi(xi,eta);
42 rt = rt / jt;
43
44 return rt;
45}

311double * RK_step1(double * un, double ** M, double * Rn, double timeStep, int d)
312{
313 double * us1 = new double [d];
314 int i, j, k, l;
315 double temp;
316
317 //*************下面计算M的逆矩阵**********//
318 double ** M_inv = new double * [d];
319 for (i=0; i<d; i++)
320 {
321 M_inv[i] = new double [d];
322 }
323 for (i=0; i<d; i++)
324 {
325 for (j=0; j<d; j++)
326 {
327 if (j == i)
328 {
329 M_inv[i][j] = 1;
330 }
331 else{
332 M_inv[i][j] = 0;
333 }
334 }
335 }
336
337 for (i=0; i<d; i++)
338 {
339 temp = M[i][i];
340 for (j=0; j<d; j++)
341 {
342 M[i][j] = M[i][j] / temp;
343 M_inv[i][j] = M_inv[i][j] / temp;
344 }
345 for (k=0; k<d; k++)
346 {
347 if ( k == i)
348 {
349 continue;
350 }
351 else{
352 temp = M[k][i];
353 for (j=0; j<d; j++)
354 {
355 M[k][j] = M[k][j] - temp * M[i][j];
356 M_inv[k][j] = M_inv[k][j] - temp * M_inv[i][j];
357 }
358 }
359 }
360 }
361 //********注意此时M已被转化为单位矩阵***********//
362
363 for (i=0; i<d; i++)
364 {
365 us1[i] = 0;
366 }
367 for (i=0; i<d; i++)
368 {
369 for (j=0; j<d; j++)
370 {
371 us1[i] = us1[i] + M_inv[i][j] * Rn[j];
372 }
373 us1[i] = us1[i] * timeStep + un[i];
374 }
375
376 for (i=0; i<d; i++)
377 {
378 delete[] M_inv[i];
379 }
380 delete[] M_inv;
381
382 return us1;
383}

482double choose_dt(double dtn)
483{
484 double timeStep;
485
486 int i, j, k, l, r;
487 double et, len, c, temp;
488
489 //************对每个单元计算CFL时间步长**********//
490 timeStep = 10000;
491 for (i=0; i<n; i++)
492 {
493 for (j=0; j<m; j++)
494 {
495 //*****声速取为质心的声速*********//
496 et = o[i][j].uxlast[0] * o[i][j].uxlast[0];
497 et = et + o[i][j].uylast[0] * o[i][j].uylast[0];
498 et = o[i][j].taulast[0] - 0.5 * et;
499
500 c = (gamma - 1) * et;
501 c = sqrt(c);
502
503 //*******下面计算最短的边长********//
504 len = 100000;
505 for (r=0; r<4; r++)
506 {
507 k = o[i][j].vertex[r] / (m+1);
508 l = o[i][j].vertex[r] % (m+1);
509
510 int node1, node2, p1, p2;
511 if (r == 3)
512 {
513 node1 = o[i][j].vertex[0];
514 }
515 else{
516 node1 = o[i][j].vertex[r+1];
517 }
518
519 if (r == 0)
520 {
521 node2 = o[i][j].vertex[3];
522 }
523 else{
524 node2 = o[i][j].vertex[r-1];
525 }
526
527 for (p1 = 0; p1 < point[k][l].neighbor_node.size(); p1++)
528 {
529 if (node1 == point[k][l].neighbor_node[p1])
530 {
531 break;
532 }
533 }
534 for (p2 = 0; p2 < point[k][l].neighbor_node.size(); p2++)
535 {
536 if (node2 == point[k][l].neighbor_node[p2])
537 {
538 break;
539 }
540 }
541
542 temp = point[k][l].a[p1] * 2;
543 if (temp < len)
544 {
545 len = temp;
546 }
547 temp = point[k][l].a[p2] * 2;
548 if (temp < len)
549 {
550 len = temp;
551 }
552 }
553
554 temp = len / c;
555
556 if (temp < timeStep)
557 {
558 timeStep = temp;
559 }
560 }
561 }
562
563 timeStep = timeStep * c_CFL;
564
565 if (timeStep > dtn)
566 {
567 timeStep = 1.01 * dtn;
568 }
569
570 return timeStep;
571}
