391double * mass_center(int i, int j)
392{
393 double * ce = new double [2];
394
395 int k;
396
397 double masst; //单元质量
398 double mf_xi, mf_eta; //质量矩
399
400 //********下面利用Gauss积分计算质量和质量矩************//
401 double rhot, jt; //记录密度和Jacobi行列式的点值
402 double xt, yt; //Gauss点的物理坐标
403 masst = 0;
404 mf_xi = 0;
405 mf_eta = 0;
406 for (k = 0; k < 4; k++)
407 {
408 xt = o[i][j].phi_x(Gausspoint_xi[k],Gausspoint_eta[k]);
409 yt = o[i][j].phi_y(Gausspoint_xi[k],Gausspoint_eta[k]);
410
411 rhot = ini_rho(xt,yt);
412 jt = o[i][j].Jacobi(Gausspoint_xi[k],Gausspoint_eta[k]);
413
414 masst = masst + rhot * jt * Gaussweight[k];
415 mf_xi = mf_xi + rhot * Gausspoint_xi[k] * jt * Gaussweight[k];
416 mf_eta = mf_eta + rhot * Gausspoint_eta[k] * jt * Gaussweight[k];
417 }
418 o[i][j].mass = masst;
419
420 //**********下面计算质心坐标***********//
421
422 ce[0] = mf_xi / masst; //质心参考空间横坐标
423 ce[1] = mf_eta / masst; //质心参考空间纵坐标
424
425 return ce;
426}

435double ** mass_matrix(int i, int j)
436{
437 int k, l;
438 double rhot, jt, xt, yt, temp;
439 double ** mma = new double * [dim];
440 for ( k = 0; k < dim; k++)
441 {
442 mma[k] = new double [dim];
443 }
444
445 for (k = 0; k<dim; k++)
446 {
447 for (l=0; l<dim; l++)
448 {
449 mma[k][l] = 0;
450 for (int r = 0; r < 4; r++)
451 {
452 xt = o[i][j].phi_x(Gausspoint_xi[r], Gausspoint_eta[r]);
453 yt = o[i][j].phi_y(Gausspoint_xi[r], Gausspoint_eta[r]);
454
455 rhot = ini_rho(xt,yt);
456
457 jt = o[i][j].Jacobi(Gausspoint_xi[r], Gausspoint_eta[r]);
458
459 temp = o[i][j].Psi(k,Gausspoint_xi[r],Gausspoint_eta[r]);
460 temp = temp * o[i][j].Psi(l,Gausspoint_xi[r],Gausspoint_eta[r]);
461
462 mma[k][l] = mma[k][l] + rhot * temp * jt * Gaussweight[r];
463 }
464 }
465 }
466
467 return mma;

474void initial()
475{
476 int i, j, k;
477 //***********下面生成网格划分**********//
478 generatemesh();
479
480 //***********下面对每个网格寻找相邻网格**********//
481 for (i=0; i<n; i++)
482 {
483 for (j=0; j<m; j++)
484 {
485 element_findneighbor(i,j);
486 }
487 }
488
489 //***********下面对每个节点寻找相邻节点***********//
490 for (i=0; i<=n; i++)
491 {
492 for (j=0; j<=m; j++)
493 {
494 node_findneighbor(i,j);
495 }
496 }
497
498 //***********下面对边界节点确定速度**************//
499 for (i=0; i<=n; i++)
500 {
501 for (j=0; j<=m; j++)
502 {
503 if (point[i][j].boundary == 1)
504 {
505 //边界节点
506 point[i][j].upstarx = ini_ux(point[i][j].x,point[i][j].y);
507 point[i][j].upstary = ini_uy(point[i][j].x,point[i][j].y);
508 }
509 else{
510 point[i][j].upstarx = 0;
511 point[i][j].upstary = 0;
512 }
513 }
514 }
515
516 //************下面计算每个网格的质量和质心，并计算质量矩阵*********//
517 for (i=0; i<n; i++)
518 {
519 for (j=0; j<m; j++)
520 {
521 //下面计算质心
522 double * ce;
523 ce = mass_center(i,j);
524 o[i][j].xi_c = ce[0];
525 o[i][j].eta_c = ce[1];
526
527 delete[] ce;
528
529 //下面计算质量矩阵
530 o[i][j].M = mass_matrix(i,j);
531 }
532 }
533
534 //***********下面给每个单元赋初值***************//
535 double xt, yt; //质心的物理坐标
536 double x_xi, x_eta, y_xi, y_eta; //Jacobi矩阵的四个分量
537 double taux, tauy;
538 for (i=0; i<n; i++)
539 {
540 for (j=0; j<m; j++)
541 {
542 o[i][j].uxlast = new double [dim];
543 o[i][j].uxnext = new double [dim];
544 o[i][j].uylast = new double [dim];
545 o[i][j].uynext = new double [dim];
546 o[i][j].taulast = new double [dim];
547 o[i][j].taunext = new double [dim];
548
549 xt = o[i][j].phi_x(o[i][j].xi_c,o[i][j].eta_c);
550 yt = o[i][j].phi_y(o[i][j].xi_c,o[i][j].eta_c);
551
552 x_xi = 0;
553 x_eta = 0;
554 y_xi = 0;
555 y_eta = 0;
556 for (k = 0; k< 4; k++)
557 {
558 x_xi = x_xi + o[i][j].vx[k] * bp_xi(k,o[i][j].xi_c,o[i][j].eta_c);
559 x_eta = x_eta + o[i][j].vx[k] * bp_eta(k,o[i][j].xi_c,o[i][j].eta_c);
560 y_xi = y_xi + o[i][j].vy[k] * bp_xi(k,o[i][j].xi_c,o[i][j].eta_c);
561 y_eta = y_eta + o[i][j].vy[k] * bp_eta(k,o[i][j].xi_c,o[i][j].eta_c);
562 }
563
564 //********下面初始化速度场************//
565 o[i][j].uxlast[0] = ini_ux(xt,yt);
566 o[i][j].uylast[0] = ini_uy(xt,yt); //在质心的值
567
568 o[i][j].uxlast[1] = ini_ux_x(xt,yt) * x_xi + ini_ux_y(xt,yt) * y_xi; //质心处ux对xi求偏导
569 o[i][j].uylast[1] = ini_uy_x(xt,yt) * x_xi + ini_ux_y(xt,yt) * y_xi; //质心处uy对xi求偏导
570
571 o[i][j].uxlast[2] = ini_ux_x(xt,yt) * x_eta + ini_ux_y(xt,yt) * y_eta; //质心处ux对eta求偏导
572 o[i][j].uylast[2] = ini_uy_x(xt,yt) * x_eta + ini_uy_y(xt,yt) * y_eta; //质心处uy对eta求偏导
573
574 //*********下面初始化总能量场*********//
575 o[i][j].taulast[0] = ini_e(xt,yt) + 0.5 * (ini_ux(xt,yt) * ini_ux(xt,yt) + ini_uy(xt,yt) * ini_uy(xt,yt));
576
577 taux = ini_e_x(xt,yt) + ini_ux(xt,yt) * ini_ux_x(xt,yt) + ini_uy(xt,yt) * ini_uy_x(xt,yt);
578 tauy = ini_e_y(xt,yt) + ini_ux(xt,yt) * ini_ux_y(xt,yt) + ini_uy(xt,yt) * ini_uy_y(xt,yt);
579
580 o[i][j].taulast[1] = taux * x_xi + tauy * y_xi;
581 o[i][j].taulast[2] = taux * x_eta + tauy * y_xi;
582 }
583 }
584
585 return ;
586}
