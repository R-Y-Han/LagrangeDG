177double Omega::Jacobi(double xi, double eta)
178{
179 double ans;
180 double a, b, c, d;
181
182 a = 0;
183 b = 0;
184 c = 0;
185 d = 0;
186 for ( int i = 0; i < 4; i++)
187 {
188 a = a + this->vx[i] * bp_xi(i,xi,eta);
189 b = b + this->vx[i] * bp_eta(i,xi,eta);
190 c = c + this->vy[i] * bp_xi(i,xi,eta);
191 d = d + this->vy[i] * bp_eta(i,xi,eta);
192 }
193 ans = a * d - b * c;
194 return ans;
195}
