75double * nodal_velocity(int i, int j)
76{
77 double* ustar = new double [2];
78
79 //**************先判断是否为边界点*********//
80 if (point[i][j].boundary == 1)
81 {
82 //******是边界点*******//
83 ustar[0] = - point[i][j].x0;
84 ustar[1] = - point[i][j].y0;
85 //边界速度不变
86 return ustar;
87 }
88
89 //*************不是边界节点***************//
90
91 int r, temp, k, l, node1, node2, p1, p2;
92 double xit, etat, nx1, ny1, nx2, ny2, a1, a2;
93 double mu, topx, topy, bottom;
94 topx = 0;
95 topy = 0;
96 bottom = 0;
97 //***************对每个corner进行计算***********//
98 for (r=0; r<point[i][j].neighbor_element.size(); r++)
99 {
100 //***********确定corner所在单元编号*********//
101 k = point[i][j].neighbor_element[r] / m;
102 l = point[i][j].neighbor_element[r] % m;
103
104 //************确定节点在单元内的位置和参考空间坐标***********//
105 for (temp=0; temp < 4; temp++)
106 {
107 if (o[k][l].vertex[temp] == point[i][j].q)
108 {
109 break;
110 }
111 }
112 xit = ref_xi[temp][0];
113 etat = ref_xi[temp][1];
114
115 //***********确定相邻节点的编号***********//
116 if (temp == 3)
117 {
118 node1 = o[k][l].vertex[0];
119 }
120 else{
121 node1 = o[k][l].vertex[temp + 1];
122 }
123
124 if (temp == 0)
125 {
126 node2 = o[k][l].vertex[3];
127 }
128 else{
129 node2 = o[k][l].vertex[temp - 1];
130 }
131 //node1, node2为相邻顶点的编号
132
133 //*********确定相邻节点和该节点的关系*********//
134 for (p1 = 0; p1 < point[i][j].neighbor_node.size(); p1++)
135 {
136 if (point[i][j].neighbor_node[p1] == node1)
137 {
138 break;
139 }
140 }
141
142 for (p2 = 0; p2 < point[i][j].neighbor_node.size(); p2++)
143 {
144 if (point[i][j].neighbor_node[p2] == node2)
145 {
146 break;
147 }
148 }
149 //p1, p2为(i,j)指向相邻顶点的指针
150
151 //***********记录单位外法向和相应长度********//
152 a1 = point[i][j].a[p1];
153 a2 = point[i][j].a[p2];
154
155 nx1 = - point[i][j].nx[p1];
156 ny1 = - point[i][j].ny[p1]; //注意此处加了负号，为使指向外向
157
158 nx2 = point[i][j].nx[p2];
159 ny2 = point[i][j].ny[p2];
160
161 //*******下面计算第r个corner中的相应分量********//
162 double xt, yt;
163 xt = o[k][l].phi_x(xit,etat);
164 yt = o[k][l].phi_y(xit,etat);
165
166 //********下面计算密度**********//
167 rho = ini_rho(xt,yt) * o[k][l].Jacobi_0(xit,etat);
168 rho = rho / o[k][l].Jacobi(xit,etat);
169
170 //********下面计算节点处速度重构值********//
171 double * uc = new double [2];
172 uc[0] = 0;
173 uc[1] = 0;
174 for (temp = 0; temp < dim; temp++)
175 {
176 uc[0] = uc[0] + o[k][l].uxlast[temp] * o[k][l].Psi(temp,xit,etat);
177 uc[1] = uc[1] + o[k][l].uylast[temp] * o[k][l].Psi(temp,xit,etat);
178 }
179
180 //********下面计算内能**********//
181
182 double ve;
183 ve = 0;
184 for (temp = 0; temp < dim; temp++)
185 {
186 ve = ve + o[k][l].taulast[temp] * o[k][l].Psi(temp,xit,etat);
187 }
188 ve = ve - 0.5 * (uc[0] * uc[0] + uc[1] * uc[1]);
189
190 //**************下面计算阻抗*************//
191 mu = (gamma - 1) * ve;
192 mu = sqrt(mu); //此时mu为声速
193 mu = rho * mu;
194
195 //*************下面计算激波单位方向*********//
196 double * e;
197 e = u_average(i,j);
198 e[0] = e[0] - uc[0];
199 e[1] = e[1] - uc[1];
200
201 double norm;
202 norm = e[0] * e[0] + e[1] * e[1];
203 norm = sqrt(norm);
204 if (norm == 0)
205 {
206 e[0] = 0;
207 e[1] = 0;
208 }
209 else{
210 e[0] = e[0] / norm;
211 e[1] = e[1] / norm;
212 }
213
214 //*********下面计算分子分母各分量********//
215 p = EOS (rho, ve);
216
217 double mid;
218 //*********下面计算第一条边的量***********//
219 mid = nx1 * e[0] + ny1 * e[1];
220 mid = mu * abs(mid) * a1;
221
222 topx = topx + mid * uc[0] + a1 * nx1 * p;
223 topy = topy + mid * uc[1] + a1 * ny1 * p;
224 bottom = bottom + mid;
225
226 //************下面计算第二条边的量*******//
227 mid = nx2 * e[0] + ny2 * e[1];
228 mid = mu * abs(mid) * a2;
229
230 topx = topx + mid * uc[0] + a2 * nx2 * p;
231 topy = topy + mid * uc[1] + a2 * ny2 * p;
232 bottom = bottom + mid;
233
234 delete[] uc;
235 delete[] e;
236 }
237 if (bottom == 0)
238 {
239 double * utemp = u_average(i,j);
240 ustar[0] = utemp[0];
241 ustar[1] = utemp[1];
242 delete[] utemp;
243 }
244 else{
245 ustar[0] = topx / bottom;
246 ustar[1] = topy / bottom;
247 }
248
249 return ustar;
250}
